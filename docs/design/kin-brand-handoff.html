Logo System

kin
kin

The mark is three circles in an equilateral triangle arrangement — two parents (bottom, r=9), one child (top, r=8). Always render with react-native-svg using the KinMark component. Never use an image file for the logo.

components/KinMark.tsx TSX 
import React from 'react';
import Svg, { Circle } from 'react-native-svg';

interface KinMarkProps {
  size?: number;
  color?: string;
}

/** The three-circle family constellation mark. Use for icons, app icon, tab bar. */
export const KinMark = ({ size = 32, color = '#7CB87A' }: KinMarkProps) => (
  <Svg width={size} height={size} viewBox="0 0 64 64">
    {/* Top circle — child */}
    <Circle cx="32" cy="20" r="8" fill={color} />
    {/* Bottom-left circle — parent */}
    <Circle cx="21.75" cy="37.9" r="9" fill={color} />
    {/* Bottom-right circle — parent */}
    <Circle cx="42.25" cy="37.9" r="9" fill={color} />
  </Svg>
);

export default KinMark;
components/KinLogo.tsx TSX 
import React from 'react';
import { View, Text, StyleSheet } from 'react-native';
import { KinMark } from './KinMark';
import { Colors, Typography } from '../constants/brand';

interface KinLogoProps {
  markSize?: number;
  fontSize?: number;
  color?: string;
  gap?: number;
}

/** Full horizontal lockup: mark + wordmark. Use in nav headers and splash. */
export const KinLogo = ({
  markSize = 24,
  fontSize = 22,
  color = Colors.text,
  gap = 10,
}: KinLogoProps) => (
  <View style={[styles.row, { gap }]}>
    <KinMark size={markSize} />
    <Text style={[styles.wordmark, { fontSize, color }]}>kin</Text>
  </View>
);

const styles = StyleSheet.create({
  row: { flexDirection: 'row', alignItems: 'center' },
  wordmark: {
    fontFamily: Typography.families.sans,
    fontWeight: Typography.weights.medium,
    letterSpacing: 0.2,
    lineHeight: 1,
  },
});

export default KinLogo;

  const [loaded] = useFonts({
    'Geist':        require('../assets/fonts/Geist-Regular.ttf'),
    'Geist-Medium':  require('../assets/fonts/Geist-Medium.ttf'),

    'GeistMono-Medium': require('../assets/fonts/GeistMono-Medium.ttf'),
  });

  if (!loaded) return null;
  SplashScreen.hideAsync();
  return <Stack />;
}

Brand Rules

RuleRequirementReason
Logo renderingAlways use the KinMark SVG component — never an image assetSVG scales perfectly at all densities; no pixelation on high-DPI screens
Logo colorMark must be #7CB87A on dark backgrounds; #5A9A58 on lightContrast compliance and brand consistency
Wordmark fontGeist weight 500, letter-spacing 0.2, lowercase onlyCustom weight/spacing is part of the brand identity
Body text fontGeist 400 for body, Geist Mono for all labels/timestampsMono establishes a visual hierarchy distinct from prose
Instrument SerifDo not use — fully removed from the brandPrevious font; all uses replaced with Geist
Background#0C0F0A for all screens; #141810 for cards/sheetsSlight green tint is intentional — not pure black
Alert colorsUse the semantic colors as defined — no substitutionsColors map to alert types; users learn the system
Border opacity7% white for default, 13% for hover/focus statesMaintains visibility without harsh contrast in dark mode
